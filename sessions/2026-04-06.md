function detectRegion(ip) {
  const geo = geoip.lookup(ip);
  
  if (!geo || !geo.country) {
    return 'global';

  const countryCode = geo.country;
  
  if (countryCode === 'TR') {
    return 'turkey';
  }
  
  if (EUROPEAN_COUNTRIES.includes(countryCode)) {

  }

}

function getClientIp(req) {
  // Proxy/Load Balancer behind
  return req.headers['x-forwarded-for']?.split(',')[0].trim() ||
         req.headers['x-real-ip'] ||
         req.connection.remoteAddress ||
         req.socket.remoteAddress ||

}

class LocationService {
  constructor() {
    this.location = null;
    this.isDetecting = false;
  }
  
  async detectLocation() {
    if (this.location) {
      return this.location; // Return cached
    }
    
    if (this.isDetecting) {
      // Wait for ongoing detection
      return new Promise((resolve) => {
        const interval = setInterval(() => {
          if (this.location) {
            clearInterval(interval);
            resolve(this.location);
          }
        }, 100);
      });
    }
    
    this.isDetecting = true;
    
    try {
      const response = await fetch('http://localhost:5000/api/detect-location');
      const data = await response.json();
      
      this.location = data;
      this.isDetecting = false;
      
      return data;
    } catch (error) {
      console.error('Location detection failed:', error);
      this.isDetecting = false;
      
      // Fallback to USD $3
      this.location = {
        region: 'global',
        pricing: { symbol: '$', formatted: '$3' }
      };
      
      return this.location;
    }
  }
  
  getLocation() {
    return this.location;
  }
  
  getPricingText() {
    if (!this.location) return 'Loading...';
    const { symbol, formatted } = this.location.pricing;
    return `${formatted}/ay`;
  }
  
  getRegion() {
    return this.location?.region || 'global';
  }
}

export default new LocationService();
```

import { supabase } from './supabaseClient';

class PaymentService {
  constructor() {
    this.API_BASE_URL = 'http://localhost:5000';
  }
  
  /**
   * Create Stripe Checkout Session
   */
  async createCheckoutSession(userId, userEmail) {
    try {
      const response = await fetch(`${this.API_BASE_URL}/api/create-checkout-session`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId, userEmail })
      });
      
      if (!response.ok) {
        throw new Error('Failed to create checkout session');
      }
      
      const data = await response.json();
      return data; // { sessionId, url }
      
    } catch (error) {
      console.error('Checkout session error:', error);
      throw error;
    }
  }
  
  /**
   * Redirect to Stripe Checkout
   */
  async redirectToCheckout(userId, userEmail) {
    try {
      const { url } = await this.createCheckoutSession(userId, userEmail);
      window.location.href = url;
    } catch (error) {
      console.error('Redirect to checkout failed:', error);
      throw error;
    }
  }
  
  /**
   * Verify payment after successful checkout
   */
  async verifyPayment(sessionId) {
    try {
      const response = await fetch(`${this.API_BASE_URL}/api/verify-payment`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ sessionId })
      });
      
      if (!response.ok) {
        throw new Error('Payment verification failed');
      }
      
      const data = await response.json();
      
      // Update Supabase license
      if (data.success) {
        await this.updateUserLicense(data.userId, data.subscription);
      }
      
      return data;
      
    } catch (error) {
      console.error('Payment verification error:', error);
      throw error;
    }
  }
  
  /**
   * Update user license in Supabase
   */
  async updateUserLicense(userId, subscription) {
    try {
      const { data, error } = await supabase
        .from('user_licenses')
        .upsert({
          user_id: userId,
          license_type: 'premium',
          is_active: true,
          stripe_subscription_id: subscription.id,
          stripe_customer_id: subscription.customer,
          subscription_status: subscription.status,
          current_period_start: new Date(subscription.current_period_start * 1000),
          current_period_end: new Date(subscription.current_period_end * 1000),
          expires_at: new Date(subscription.current_period_end * 1000),
          updated_at: new Date()
        }, {
          onConflict: 'user_id'
        });
      
      if (error) throw error;
      
      return data;
      
    } catch (error) {
      console.error('License update error:', error);
      throw error;
    }
  }
  
  /**
   * Get user subscription status
   */
  async getSubscriptionStatus(userId) {
    try {
      const { data, error } = await supabase
        .from('user_licenses')
        .select('*')
        .eq('user_id', userId)
        .single();
      
      if (error) {
        if (error.code === 'PGRST116') {
          // No license found
          return null;
        }
        throw error;
      }
      
      return data;
      
    } catch (error) {
      console.error('Subscription status error:', error);
      return null;
    }
  }
}

export default new PaymentService();

import { useEffect, useState } from 'react';
import { useSearchParams, useNavigate } from 'react-router-dom';
import paymentService from '../services/paymentService';
import useAuthStore from '../store/authStore';
import { CheckCircle, Loader, XCircle } from 'lucide-react';

const PaymentSuccessPage = () => {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const refreshUser = useAuthStore(state => state.refreshUser);
  
  const [status, setStatus] = useState('verifying'); // 'verifying' | 'success' | 'error'
  const [countdown, setCountdown] = useState(3);
  
  useEffect(() => {
    const verifyAndRedirect = async () => {
      const sessionId = searchParams.get('session_id');
      
      if (!sessionId) {
        setStatus('error');
        return;
      }
      
      try {
        // Verify payment
        await paymentService.verifyPayment(sessionId);
        
        // Refresh user auth state (premium status)
        await refreshUser();
        
        setStatus('success');
        
        // Countdown timer
        const interval = setInterval(() => {
          setCountdown(prev => {
            if (prev <= 1) {
              clearInterval(interval);
              navigate('/dashboard');
              return 0;
            }
            return prev - 1;
          });
        }, 1000);
        
      } catch (error) {
        console.error('Payment verification failed:', error);
        setStatus('error');
      }
    };
    
    verifyAndRedirect();
  }, [searchParams, navigate, refreshUser]);
  
  return (
    <div className="min-h-screen flex items-center justify-center">
      {status === 'verifying' && (
        <div className="text-center">
          <Loader className="w-16 h-16 animate-spin" />
          <p>Ödemeniz doğrulanıyor...</p>
        </div>
      )}
      
      {status === 'success' && (
        <div className="text-center">
          <CheckCircle className="w-24 h-24 text-green-500" />
          <h1>Ödeme Başarılı!</h1>
          <p>Premium hesabınız aktif edildi</p>
          <p>{countdown} saniye sonra yönlendiriliyorsunuz...</p>
        </div>
      )}
      
      {status === 'error' && (
        <div className="text-center">
          <XCircle className="w-24 h-24 text-red-500" />
          <h1>Ödeme Doğrulanamadı</h1>
          <button onClick={() => navigate('/')}>
            Ana Sayfaya Dön
          </button>
        </div>
      )}
    </div>
  );
};

export default PaymentSuccessPage;
```

**Akış:**
1. URL'den `session_id` parametresi okunur
2. `paymentService.verifyPayment(sessionId)` çağrılır
3. Backend Stripe'dan session bilgilerini çeker
4. Supabase `user_licenses` tablosu güncellenir
5. User auth state refresh edilir (premium status)
6. 3 saniyelik countdown
7. `/dashboard` sayfasına yönlendirilir
